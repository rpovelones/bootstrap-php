Bootstrap Tab Examples

Default


<?php
$tabs = new \RPD\Bootstrap\Tab_Section('my_tab_section', [
  [
    'title'  => 'Tab One',
    'slug'   => 'tab-1',
  ],
  [
    'title'  => 'Tab Two',
    'slug'   => 'tab-2',
  ]
]);
?>

<nav class="my-nav">
  <?php $tabs->output_nav(); ?>
</nav>

<?php $tabs->output_content(); ?>

Pill Tabs with Fade


<?php
$settings = [
  'tab_nav_class' => 'nav-pills',
  'fade' => true
];

$tabs = new \RPD\Bootstrap\Tab_Section('my_tab_section', [
  [
    'title'  => 'Tab One',
    'slug'   => 'tab-1',
  ],
  [
    'title'  => 'Tab Two',
    'slug'   => 'tab-2',
  ]
], $settings);
?>

<nav class="my-nav">
  <?php $tabs->output_nav(); ?>
</nav>

<?php $tabs->output_content(); ?>

Vertical Tabs


<?php
$settings = [
  'tab_nav_class' => 'nav-pills flex-column',
  'fade' => true
];

$tabs = new \RPD\Bootstrap\Tab_Section('my_tab_section', [
  [
    'title'  => 'Tab One',
    'slug'   => 'tab-1',
  ],
  [
    'title'  => 'Tab Two',
    'slug'   => 'tab-2',
  ]
], $settings);
?>

<div class="row">
  <div class="col-3">
    <nav class="my-nav">
      <?php $tabs->output_nav(); ?>
    </nav>
  </div>
  <div class="col-9">
    <?php $tabs->output_content(); ?>
  </div>
</div>
